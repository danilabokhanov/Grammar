\section{Gerund and infinitive}
\subsection {Общие правила}
\begin{enumerate}
    \item После предлогов всегда \bld{ing}
    \item После существительных всегда \bld{ing}
    \item После прилагательных \bld{infinitive}
    \item После вопросительных слов \bld{infinitive}
\end{enumerate}

\subsection {Фразы и глаголы с infinitive}
\p
\sample{decide, seem, pass, promise, remember\bld{*}$_1$, offer, manage, would you like, prefer\bld{*}$_2$, expect, appear,
refuse, learn, used to, agree, likely to, encourage, warn, advise, forget, remind, take, watch, pretend, see,
intend, be able, beg, try\bld{*}$_3$, help, force, continue\bld{*}$_4$, teach, like}
\bld{*}$_1$ не забудь что-то сделать, но я помню, как я что-то сделал, 
будет с \bld{ing}\\
\bld{*}$_2$ вариант с \bld{ing} тоже используется\\
\bld{*}$_3$ см prefer\\
\bld{*}$_4$ см prefer

\subsection {Фразы и глаголы с ing}
\p
\sample{imagine, enjoy, mind, remember\bld{*}$_1$, prefer\bld{*}$_2$, keep, couldn't help (я не мог не), give up, stop, deny, admit, hate,
avoid, try\bld{*}$_3$, continue\bld{*}$_4$, like}
\bld{*}$_1$ в контексте "я помню, как я что-то сделал"\\
\bld{*}$_2$ вариант с \bld{infinitive} тоже используется\\
\bld{*}$_3$ см prefer\\
\bld{*}$_4$ см prefer

\subsection {Фразы и глаголы без всего}
\p
\sample{why not, let, would rather, make, couldn't but (я случайно, я не мог не), explode, better}

\subsection {Фразы и глаголы с to + ing}
\p
\sample{looking forward to, to be used to + ing (мне привычно), get used to}

\subsection {Примеры}

\sample{It is not worth to help him to do the job}

\sample{Her forces us to accept his offer by raising it by 5 \%}
