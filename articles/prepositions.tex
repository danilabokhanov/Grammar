\section{Prepositions}
\subsection {глаголы с предлогом}
\begin{description}
    \item \bld{accuse sb of sth} (обвинять кого-то в чём-то)
    \item \bld{arrive in} (пребывать в города, страны)
    \item \bld{arrive at} (пребывать в аэропорт, на станцию, в университет)
    \item \bld{(dis)agree with sb} ((не) cогласиться с кем-то)
    \item \bld{aim at sth} (целиться во что-то)
    \item \bld{apologise to sb for sth} (извеняться перед кем-то за что-то)
    \item \bld{apply to sb for sth} (обратиться к кому-то за чем-то)
    \item \bld{beg for sth} (умолять о чём-то)
    \item \bld{believe in sth} (верить во что-то)
    \item \bld{belong to sb/sth} (принадлежать кому-то/чему-то)
    \item \bld{benefit from sth} (извлекать выгоду из чего-то)
    \item \bld{beware of sb/sth} (остерегаться кого-то/чего-то)
    \item \bld{care about sb/sth} (беспокоиться/заботиться о ком-то/чём-то)
    \item \bld{care for sb/sth} (заботиться/ухаживать за кем-то/чем-то)
    \item \bld{choose from/between} (выбирать из/между)
    \item \bld{collaborate with} (сотрудничать с)
    \item \bld{combine with} (сочетать с)
    \item \bld{comment on sb/sth} (комметировать кого-то/что-то)
    \item \bld{communicate with} (общаться с)
    \item \bld{compare with/to} (сравнивать с)
    \item \bld{compete with sb for sth} (конкурировать с кем-то за что-то)
    \item \bld{concentrate on sth} (концентрироваться на чём-то)
\end{description}
